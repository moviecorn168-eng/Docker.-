import std/[os, strutils, db_sqlite, strformat]
import models

const
  DataRoot = "data/databases"

proc sanitizeName*(name: string): string =
  result = ""
  for c in name:
    if c in {'a'..'z', 'A'..'Z', '0'..'9', '_', '-'}:
      result.add c
  if result.len == 0:
    raise newException(ValueError, "Database name must contain at least one alphanumeric character")
  if result.len > 63:
    result = result[0..62]

proc ensureDataRoot() =
  if not dirExists(DataRoot):
    createDir(DataRoot)

proc provisionSqlite*(ownerId, safeName: string): string =
  ensureDataRoot()

  let ownerDir = DataRoot / ownerId
  if not dirExists(ownerDir):
    createDir(ownerDir)

  let dbPath = ownerDir / (safeName & ".sqlite3")
  if fileExists(dbPath):
    raise newException(IOError, fmt"Database '{safeName}' already exists for this owner")

  let conn = open(dbPath, "", "", "")
  try:
    conn.exec(sql"""
      CREATE TABLE IF NOT EXISTS _meta (
        key   TEXT PRIMARY KEY,
        value TEXT
      )
    """)
    conn.exec(sql"INSERT INTO _meta (key, value) VALUES (?, ?)",
               "provisioned_by", "mini-render-control-plane")
  finally:
    conn.close()

  result = dbPath

proc deprovisionSqlite*(dbPath: string) =
  if fileExists(dbPath):
    removeFile(dbPath)

import std/db_postgres

proc provisionPostgres*(safeName: string): string =
  let
    host = getEnv("PG_ADMIN_HOST", "localhost")
    user = getEnv("PG_ADMIN_USER", "postgres")
    pass = getEnv("PG_ADMIN_PASSWORD", "")
    adminDb = getEnv("PG_ADMIN_DB", "postgres")

  let conn = db_postgres.open(host, user, pass, adminDb)
  try:
    conn.exec(sql(fmt"""CREATE DATABASE "{safeName}""""))
  finally:
    conn.close()

  result = fmt"postgresql://{user}@{host}/{safeName}"

proc provisionDatabase*(ownerId, rawName: string, engine: DatabaseEngine): string =
  let safeName = sanitizeName(rawName)
  case engine
  of dbSqlite:
    result = provisionSqlite(ownerId, safeName)
  of dbPostgres:
    result = provisionPostgres(safeName)
